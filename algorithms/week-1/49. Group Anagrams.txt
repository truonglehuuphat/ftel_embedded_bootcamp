Given an array of strings strs, group the anagrams together. You can return the answer in any order.

 

Example 1:

Input: strs = ["eat","tea","tan","ate","nat","bat"]

Output: [["bat"],["nat","tan"],["ate","eat","tea"]]

Explanation:

There is no string in strs that can be rearranged to form "bat".
The strings "nat" and "tan" are anagrams as they can be rearranged to form each other.
The strings "ate", "eat", and "tea" are anagrams as they can be rearranged to form each other.
Example 2:

Input: strs = [""]

Output: [[""]]

Example 3:

Input: strs = ["a"]

Output: [["a"]]

 

Constraints:

1 <= strs.length <= 10^4
0 <= strs[i].length <= 100
strs[i] consists of lowercase English letters.

class Solution {
public:
    static bool lengthCompareArray(vector<string>&a, vector<string>& b) {
        return a.size() < b.size();
    }
    static bool lengthCompare(string &a, string &b) {
        return a.length() < b.length(); 
    }
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        int array[10001] = {0};
        vector<vector<string>> result;
        int l = 0;
        for (int r = 0; r < strs.size(); r++){ 
            if(array[r] == 0){
                array[r] = 1;
                vector<string> temp;
                temp.push_back(strs[r]);
                for(l = 0; l < strs.size(); l++){
                    if (array[l] == 0 && (strs[r].size() == strs[l].size())){
                        int hashtable[26] = {0};
                        int flag = 0;
                        for(char c : strs[r]){
                            hashtable[c-'a']++;
                        }
                        for(char c: strs[l]){
                            if(hashtable[c-'a'] == 0){
                                flag = 1;
                                break;
                            }
                            hashtable[c-'a']--;
                        }
                        if(flag == 0){
                            array[l] = 1;
                            temp.push_back(strs[l]);
                            sort(temp.begin(), temp.end());
                        }
                    }
                }
                result.push_back(temp);
                temp.clear();
            }
        }

        sort(result.begin(), result.end(), lengthCompareArray);
        return result;
    }
};

class Solution {
    public List<List<String>> groupAnagrams(String[] strs) {
        Map<String,List<String>> map = new HashMap<>();
        for(String  x: strs){
            char[] chars = x.toCharArray();
            Arrays.sort(chars);
            String sortWord = new String(chars);
            if(!map.containsKey(sortWord)){
                map.put(sortWord, new ArrayList<>());
            }
            map.get(sortWord).add(x);
        }
        return new ArrayList<>(map.values());
    }
}


class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        map<string, vector<string>> map;
        for(auto word : strs){
           string sortWord = word;
           sort(sortWord.begin(),sortWord.end());
           map[sortWord].push_back(word);
        }

        vector<vector<string>> ans;
        for (auto x: map){
            ans.push_back(x.second);
        }
        return ans;
    }
};
